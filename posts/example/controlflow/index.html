Go 流程控制

条件控制 if-else

// if-else基本例子
if 8%2 == 0 {
    fmt.Println("8是偶数")
} else {
    fmt.Println("8是奇数")
}
// 条件之前带语句的 if-else
if i := 5; i < 0 {
    fmt.Println(i, "为负数")
} else if i < 10 {
    fmt.Println(i, "为1位数")
} else {
    fmt.Println(i, "为多位数")
}

分支结构 switch

// switch
sex := 1
switch sex {
case 0:
    fmt.Println("保密")
case 1:
    fmt.Println("男")
case 2:
    fmt.Println("女")
}
// 一个分支中多表达式
switch time.Now().Weekday() {
case time.Saturday, time.Sunday:
    fmt.Println("周末了，出去浪")
default:
    fmt.Println("工作日，努力搬砖")
}
// 不带表达式的switch
t := time.Now()
switch {
case t.Hour() < 12:
    fmt.Println("上午")
default:
    fmt.Println("下午")
}
